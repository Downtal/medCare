@ExtendWith(MockitoExtension.class)
class ShippingServiceTest {

    @Mock private ShipmentRepository shipmentRepository;
    @Mock private RestTemplate restTemplate;
    @InjectMocks private ShippingService shippingService;

    // Test 1: shouldCalculateShippingFeeSuccessfully
    @Test
    void shouldCalculateShippingFeeSuccessfully() {
        // Given: mock GHN API response with fee = 30000
        ResponseEntity<Map> mockResponse = ResponseEntity.ok(Map.of("data", Map.of("total", 30000)));
        when(restTemplate.exchange(anyString(), eq(HttpMethod.POST), any(), eq(Map.class)))
            .thenReturn(mockResponse);
        
        // When: calculateShippingFee() called
        // Then: returns 30000
    }

    // Test 2: shouldThrowExceptionWhenGHNAPIFails
    @Test
    void shouldThrowExceptionWhenGHNAPIFails() {
        // Given: GHN returns 500
        when(restTemplate.exchange(...)).thenThrow(new RestClientException("GHN unavailable"));
        
        // When/Then: throws RuntimeException
        assertThrows(RuntimeException.class, () -> shippingService.calculateShippingFee(...));
    }

    // Test 3: shouldCreateShipmentWithCorrectData
    @Test
    void shouldCreateShipmentWithCorrectData() {
        // Given: order data + mock GHN create order response
        // When: createShipment() called
        // Then: Shipment saved with tracking code from GHN response
    }

    // Test 4: shouldUpdateShipmentStatusWhenWebhookReceived
    @Test
    void shouldUpdateShipmentStatusWhenWebhookReceived() {
        // Given: Existing shipment + webhook payload with status DELIVERED
        // When: processWebhook() called
        // Then: shipment.status = DELIVERED, saved
    }

    // Test 5: shouldHandleUnknownWebhookStatusGracefully
    @Test
    void shouldHandleUnknownWebhookStatusGracefully() {
        // Given: webhook with unknown status code
        // When: processWebhook() called
        // Then: no exception, logs warning, status remains unchanged
    }
}
