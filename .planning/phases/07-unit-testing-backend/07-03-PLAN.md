@ExtendWith(MockitoExtension.class)
class PromotionServiceTest {

    @Mock private VoucherRepository voucherRepository;
    @Mock private VoucherUsageRepository voucherUsageRepository;
    @Mock private UserVoucherRepository userVoucherRepository;
    @InjectMocks private PromotionService promotionService;

    // Test 1: shouldApplyVoucherSuccessfullyWhenValid
    @Test
    void shouldApplyVoucherSuccessfullyWhenValid() {
        // Given: valid, unused voucher with discount 50k, order total 200k
        Voucher voucher = buildVoucher("SAVE50", 50000, false, LocalDateTime.now().plusDays(7));
        when(voucherRepository.findByCode("SAVE50")).thenReturn(Optional.of(voucher));
        when(voucherUsageRepository.countByVoucherCode("SAVE50")).thenReturn(0L);
        
        // When
        ApplyVoucherResponse result = promotionService.applyVoucher("SAVE50", "user1", 200000);
        
        // Then
        assertTrue(result.isSuccess());
        assertEquals(50000, result.getDiscountAmount());
    }

    // Test 2: shouldRejectVoucherWhenExpired
    @Test
    void shouldRejectVoucherWhenExpired() {
        // Given: voucher expired yesterday
        Voucher expired = buildVoucher("OLD50", 50000, false, LocalDateTime.now().minusDays(1));
        when(voucherRepository.findByCode("OLD50")).thenReturn(Optional.of(expired));
        
        // When
        ApplyVoucherResponse result = promotionService.applyVoucher("OLD50", "user1", 200000);
        
        // Then
        assertFalse(result.isSuccess());
        assertTrue(result.getMessage().contains("hết hạn") || result.getMessage().contains("expired"));
    }

    // Test 3: shouldRejectVoucherWhenUsageLimitReached
    @Test
    void shouldRejectVoucherWhenUsageLimitReached() {
        // Given: voucher max_usage=100, current_usage=100
        Voucher voucher = buildVoucher("LIMIT100", 30000, false, LocalDateTime.now().plusDays(7));
        voucher.setMaxUsage(100);
        when(voucherRepository.findByCode("LIMIT100")).thenReturn(Optional.of(voucher));
        when(voucherUsageRepository.countByVoucherCode("LIMIT100")).thenReturn(100L);
        
        // When/Then
        assertFalse(promotionService.applyVoucher("LIMIT100", "user1", 200000).isSuccess());
    }

    // Test 4: shouldRejectNonExistentVoucher
    @Test
    void shouldRejectNonExistentVoucher() {
        when(voucherRepository.findByCode("FAKE")).thenReturn(Optional.empty());
        
        ApplyVoucherResponse result = promotionService.applyVoucher("FAKE", "user1", 100000);
        assertFalse(result.isSuccess());
    }

    @Test

## Task 3: Cấu hình JaCoCo Coverage Report

### Thêm JaCoCo vào root build.gradle (hoặc từng service):

```gradle
// BE/build.gradle (root level)
plugins {
    id 'jacoco'
}

subprojects {
    apply plugin: 'jacoco'
    
    jacocoTestReport {
        reports {
            xml.required = true
            html.required = true
        }
        dependsOn test
    }
}
```

### Lệnh generate report:
```bash
gradlew order-service:jacocoTestReport
gradlew payment-service:jacocoTestReport
gradlew shipping-service:jacocoTestReport
gradlew promotion-service:jacocoTestReport
```

Report sẽ được tạo tại: `{service}/build/reports/jacoco/test/html/index.html`
